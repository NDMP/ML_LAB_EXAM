# Step 1: Import libraries
import pandas as pd
from sklearn.preprocessing import LabelEncoder
from sklearn.tree import DecisionTreeClassifier

# Step 2: Load data
df = pd.read_csv("salaries.csv")

# Step 3: Separate features and target
X = df[['company', 'job', 'degree']]
y = df['salary_more_then_100k']

# Step 4: Label encode categorical columns
le = LabelEncoder()
X['company_n'] = le.fit_transform(X['company'])
X['job_n'] = le.fit_transform(X['job'])
X['degree_n'] = le.fit_transform(X['degree'])

X_n = X[['company_n', 'job_n', 'degree_n']]

# Step 5: Train model
model = DecisionTreeClassifier()
model.fit(X_n, y)

# Step 6: Accuracy
print("Accuracy:", model.score(X_n, y))

# Step 7: Predictions
print("Google, Computer Programmer, Bachelors:",
      model.predict([[2, 1, 0]]))

print("Google, Computer Programmer, Masters:",
      model.predict([[2, 1, 1]]))
