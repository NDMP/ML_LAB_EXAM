# Step 1: Import pandas
import pandas as pd

# Step 2: Create DataFrame
data = pd.DataFrame({
    "A": [1, 2, 3, 4, 5, 6],
    "B": [7, 8, 9, 10, 11, 12],
    "C": [0, 0, 0, 0, 0, 0],
    "D": [21, 54, 32, 85, 35, 2]
})

print("Original DataFrame:\n", data)

# Step 3: Import VarianceThreshold
from sklearn.feature_selection import VarianceThreshold

# Step 4: Apply Variance Threshold (remove zero variance features)
var_thres = VarianceThreshold(threshold=0)
var_thres.fit(data)

# Step 5: Identify constant (zero variance) columns
constant_columns = [column for column in data.columns if column not in data.columns[var_thres.get_support()]]

print("\nConstant Columns:")
for feature in constant_columns:
    print(feature)

# Step 6: Drop constant columns
data = data.drop(constant_columns, axis=1)

print("\nDataFrame after removing constant columns:\n", data)
